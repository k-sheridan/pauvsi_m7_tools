1function [ Trajectory, totalFlightTime ] = minimumTimeTrajectoryGenerator( start, mid, final, MIDPOINT_MODE, Mass, Moment, MaxVel, MinZForce, MaxForce, MaxAngle)
 2%UNTITLED7 Summary of this function goes here
 3%   This will generate a trajectory that is flable by a quadrotor it may
 4%   take a second.
 5%   The format for start and end is [x_0, x_1, x_2, x_3, x_4]
 6%                                   [y_0, y_1, y_2, y_3, y_4]
 7%                                   [z_0, z_1, z_2, z_3, z_4]
 8%
 9%   the format for mid is [x_0, x2_0, ...]    [x_0, x2_0]
10%                         [y_0, y2_0, ...]    [y_0, y2_0]
11%                         [z_0, z2_0, ...]    [z_0, z2_0]
12%                         [x_1, x2_1, ...]
13%                         [y_1, y2_1, ...]
14%                         [z_1, z2_1, ...]
15%                    MODE:     'VEL'            'NO_VEL'    'NONE'
16%                   
17
18%define Trajectory
19[~, midCols] = size(mid);
20Trajectory = zeros(3, 11, midCols + 1);
21
22%if this is in velocity mode
23if strcmp(MIDPOINT_MODE, 'VEL') == 1
24    
25    Trajectory(:, :, 1) = polynomialTrajectorySolver([start(1, :), mid(1, 1), mid(4, 1), 0, 0, 0], [start(2, :), mid(2, 1), mid(5, 1), 0, 0, 0], [start(3, :), mid(3, 1), mid(6, 1), 0, 0, 0], Mass, Moment, MaxVel, MinZForce, MaxForce, MaxAngle);
26    
27    for z_index = (1:1:midCols);
28       if ~(z_index == midCols)
29           Trajectory(:, :, z_index + 1) = polynomialTrajectorySolver([mid(1, z_index), mid(4, z_index), 0, 0, 0, mid(1, z_index + 1), mid(4, z_index + 1), 0, 0, 0], [mid(2, z_index), mid(5, z_index), 0, 0, 0, mid(2, z_index + 1), mid(5, z_index + 1), 0, 0, 0], [mid(3, z_index), mid(6, z_index), 0, 0, 0, mid(3, z_index + 1), mid(6, z_index + 1), 0, 0, 0], Mass, Moment, MaxVel, MinZForce, MaxForce, MaxAngle);
30       else
31           Trajectory(:, :, z_index + 1) = polynomialTrajectorySolver([mid(1, z_index), mid(4, z_index), 0, 0, 0, final(1, :)], [mid(2, z_index), mid(5, z_index), 0, 0, 0, final(2, :)], [mid(3, z_index), mid(6, z_index), 0, 0, 0, final(3, :)], Mass, Moment, MaxVel, MinZForce, MaxForce, MaxAngle);
32       end
33    end
34    
35elseif strcmp(MIDPOINT_MODE, 'NO_VEL') == 1
36    
37else
38    
39end
40
41totalFlightTime = sum(Trajectory(1, 11, :));
42
43%plot the trajectory
44%[p1, p2] = trajectoryPlotter(Trajectory);
45    
46%daspect([5 5 5])
47%axis([-1 11 -1 11 -10 10])
48%hold on
49%arrow3(p1, p2, 'b', 0.4)
50%hold off
51
52end
